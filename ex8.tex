level 2/.style={sibling distance=7em}, level distance=4cm]

  child { node {} \myname{\textcolor{#2}{no bribe}}{\textcolor{#2}{$b=0$}} }

    child {node {} \myname{silently}{refuse} }
    child {node {} \myname{silently}{pay} }
    child {node {} \myname{\textcolor{#1}{and report}}{\textcolor{#1}{pay}} }
    \myname{bribe}{$b>0$}
  };
\end{tikzpicture}
\end{center}
\end{frame}
}
